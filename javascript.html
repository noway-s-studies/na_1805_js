3. Elágazások

3.1. if, else

    let feltetel = true;
    if ( feltetel ) {
      console.log('a feltétel igaz');
    } else {
      console.log('a feltétel hamis');
    }

3.2. if, else if, else

    let szerepkor = 'felhasznalo'; // lehetne még 'admin'
    if ( szerepkor == 'admin' ) {
      console.log('Üdvözöllek kedves admin!');
    } else if (szerepkor == 'felhasznalo'){
      console.log('Üdvözöllek kedves felhasználó!');
    } else {
      console.log('Üdvözöllek kedves idegen!');
    }

3.3. switch, case

    let szerepkor = 'vendeg'; // lehetne még 'admin', 'felhasznalo'
    switch ( szerepkor ) {
      case 'admin':
        console.log('Üdvözöllek kedves admin!');

      case 'felhasznalo':
        console.log('Üdvözöllek kedves felhasználó!');
      
      case 'vendeg':
        console.log('Üdvözöllek kedves vendég!');
    }
    
    let szerepkor = 'admin'; // lehetne még 'felhasznalo', 'vendeg'
    switch ( szerepkor ) {
      case 'admin':
        console.log('Üdvözöllek kedves admin!')
        break;

      case 'felhasznalo':
        console.log('Üdvözöllek kedves felhasználó!');
        break;
      
      case 'vendeg':
        console.log('Üdvözöllek kedves vendég!');
        break;

      default:
        console.log('Üdvözöllek kedves idegen!');
        break;
    }

4. Ciklusok

4.1. for

    let szoveg = 'alma körte labda';
    let aSzamlalo = 0;
    for ( let index = 0; index < szoveg.length; index++ ) {
      if(szoveg.charAt(index) == 'a'){
        aSzamlalo++;
      }
    }
    console.log('aSzamlalo: ', aSzamlalo);

4.2. while

    let szoveg = 'alma körte labda';
    let aSzamlalo = 0;
    let index = 0;
    while ( index < szoveg.length ) {
      
      if(szoveg.charAt(index) == 'a'){
        aSzamlalo++;
      }
      
      index++;
    }
    console.log('aSzamlalo: ', aSzamlalo);
